118    def charlines_in_area(self, area: Rectangle,
119                          predicate: Callable[[Character], bool] = None,
120                          rtol: float = None) -> list[CharLine]:
121        """
122        Coalesce the characters in the area and predicate into lines.
123
124        1. Every character in the area is filtered by the `predicate`.
125        2. Character orientation is split into horizontal (left->right) and
126           vertical (bottom->top) character lines sorted by x or y position.
127           Lines containing only whitespace are discarded.
128        3. Overlapping character lines are merged into sub- and superscript
129           using `rtol * max(current_line.height, next_line.height)` as the
130           tolerance for checking if the lines overlap.
131        4. The characters in the merged lines are re-sorted by origin.
132
133        :param area: Area to search for characters.
134        :param predicate: Function to discard characters in the area or include all by default.
135        :param rtol: Relative tolerance to separate lines vertically or use `sc` spacing by default.
136        :return: A list of character lines sorted by x or y position.
137        """
138        if rtol is None: rtol = self._spacing["sc"]
139        # Split all chars into lines based on rounded origin
140        origin_lines_y = defaultdict(list)
141        origin_lines_x = defaultdict(list)
142        for char in self.chars_in_area(area):
143            # Ignore all characters we don't want
144            if predicate is not None and not predicate(char):
145                continue
146            cunicode = self._unicode_filter(char.unicode)
147            if cunicode is None: continue
148            char.unicode = cunicode
149            if char.unicode < 32 and char.unicode not in {0xa}:
150                continue
151            # Ignore characters without width that are not spaces
152            if not char.width and char.unicode not in {0xa, 0xd, 0x20}:
153                _LOGGER.error(f"Unknown char width for {char}: {char.bbox}")
154            # Split up the chars depending on the orientation
155            if 45 < char.rotation <= 135 or 225 < char.rotation <= 315:
156                origin_lines_x[round(char.origin.x, 1)].append(char)
157            elif char.rotation <= 45 or 135 < char.rotation <= 225 or 315 < char.rotation:
158                origin_lines_y[round(char.origin.y, 1)].append(char)
159            else:
160                _LOGGER.error("Unknown char rotation:", char, char.rotation)
161
162        # Convert characters into lines
163        bbox_lines_y = []
164        for chars in origin_lines_y.values():
165            # Remove lines with whitespace only
166            if all(c.unicode in {0xa, 0xd, 0x20} for c in chars):
167                continue
168            origin = statistics.fmean(c.origin.y for c in chars)
169            line = CharLine(self, chars,
170                            min(c.bbox.bottom for c in chars),
171                            origin,
172                            max(c.bbox.top for c in chars),
173                            max(c.height for c in chars),
174                            sort_origin=self.height - origin)
175            bbox_lines_y.append(line)
176            # print(line, line.top, line.origin, line.bottom, line.height)
177        bbox_lines = sorted(bbox_lines_y, key=lambda l: l._sort_origin)
178
179        bbox_lines_x = []
180        for chars in origin_lines_x.values():
181            # Remove lines with whitespace only
182            if all(c.unicode in {0xa, 0xd, 0x20} for c in chars):
183                continue
184            line = CharLine(self, chars,
185                            min(c.bbox.left for c in chars),
186                            statistics.fmean(c.origin.x for c in chars),
187                            max(c.bbox.right for c in chars),
188                            max(c.width for c in chars),
189                            270 if sum(c.rotation for c in chars) <= 135 * len(chars) else 90)
190            bbox_lines_x.append(line)
191        bbox_lines += sorted(bbox_lines_x, key=lambda l: l._sort_origin)
192
193        if not bbox_lines:
194            return []
195
196        # Merge lines that have overlapping bbox_lines
197        # FIXME: This merges lines that "collide" vertically like in formulas
198        merged_lines = []
199        current_line = bbox_lines[0]
200        for next_line in bbox_lines[1:]:
201            height = max(current_line.height, next_line.height)
202            # Calculate overlap via normalize origin (increasing with line index)
203            if ((current_line._sort_origin + rtol * height) >
204                (next_line._sort_origin - rtol * height)):
205                # if line.rotation or self.rotation:
206                #     # The next line overlaps this one, we merge the shorter line
207                #     # (typically super- and subscript) into taller line
208                #     use_current = len(current_line.chars) >= len(next_line.chars)
209                # else:
210                use_current = current_line.height >= next_line.height
211                line = current_line if use_current else next_line
212                current_line = CharLine(self, current_line.chars + next_line.chars,
213                                        line.bottom, line.origin, line.top,
214                                        height, line.rotation,
215                                        sort_origin=line._sort_origin)
216            else:
217                # The next line does not overlap the current line
218                merged_lines.append(current_line)
219                current_line = next_line
220        # append last line
221        merged_lines.append(current_line)
222
223        # Sort all lines horizontally based on character origin
224        sorted_lines = []
225        for line in merged_lines:
226            if line.rotation == 90:
227                def sort_key(char):
228                    if char.unicode in {0xa, 0xd}:
229                        return char.tbbox.midpoint.y - 1e9
230                    return char.tbbox.midpoint.y
231            elif line.rotation == 270:
232                def sort_key(char):
233                    if char.unicode in {0xa, 0xd}:
234                        return -char.tbbox.midpoint.y + 1e9
235                    return -char.tbbox.midpoint.y
236            else:
237                def sort_key(char):
238                    if char.unicode in {0xa, 0xd}:
239                        return char.origin.x + 1e9
240                    return char.origin.x
241            sorted_lines.append(CharLine(self, sorted(line.chars, key=sort_key),
242                                         line.bottom, line.origin,
243                                         line.top, line.height,
244                                         line.rotation, area.left,
245                                         sort_origin=line._sort_origin))
246
247        return sorted_lines

249    def graphic_bboxes_in_area(self, area: Rectangle, with_graphics: bool = True) -> list[tuple[Rectangle, Table | Figure | None]]:
250        """
251        Coalesce the graphics in the area into full width bounding boxes.
252
253        1. Group vertically overlapping graphics.
254        2. Widen the overlapped graphics bounding boxes to the edges of the area.
255
256        :param area: area to search for content.
257        :param with_graphics: search for graphics in the area.
258        :return: list of tuples (bounding box, graphic objects or `None`).
259        """
260        if with_graphics:
261            graphics = self.graphics_in_area(area)
262            regions = []
263            # Check if graphics bounding boxes overlap vertically and group them
264            for graphic in sorted(graphics, key=lambda g: (-g.bbox.top, g.bbox.x)):
265                gbbox = graphic.bbox.joined(graphic.cbbox) if graphic.cbbox else graphic.bbox
266                for reg in regions:
267                    if reg.overlaps(gbbox.bottom, gbbox.top):
268                        # They overlap, so merge them
269                        reg.v0 = min(reg.v0, gbbox.bottom)
270                        reg.v1 = max(reg.v1, gbbox.top)
271                        reg.objs.append(graphic)
272                        break
273                else:
274                    regions.append(Region(gbbox.bottom, gbbox.top, graphic))
275
276            # print(regions)
277            # Coalesce all overlapped graphics objects into full width areas
278            areas = []
279            ypos = area.top
280            for reg in regions:
281                if ypos - reg.v1 > self._spacing["y_em"]:
282                    areas.append((Rectangle(area.left, reg.v1, area.right, ypos), None))
283                for obj in reg.objs:
284                    oarea = obj.bbox.joined(obj.cbbox) if obj.cbbox else obj.bbox
285                    areas.append((oarea, obj))
286                ypos = reg.v0
287            areas.append((Rectangle(area.left, area.bottom, area.right, ypos), None))
288        else:
289            areas = [(area, None)]
290        return areas

292    def objects_in_area(self, area: Rectangle, with_graphics: bool = True) -> list[CharLine | Table | Figure]:
293        """
294        Find all content objects in this area.
295
296        :param area: area to search for content.
297        :param with_graphics: search for graphics in the area.
298        :return: list of content objects sorted top to bottom.
299        """
300        self._link_characters()
301        areas = self.graphic_bboxes_in_area(area, with_graphics)
302        objects = []
303        for narea, obj in areas:
304            if obj is None:
305                objects += self.charlines_in_area(narea)
306            else:
307                oarea = obj.bbox.joined(obj.cbbox) if obj.cbbox else obj.bbox
308                predicate = lambda c: not obj.bbox.contains(c.origin)
309                lines = self.charlines_in_area(oarea, predicate)
310                # print(obj, oarea, lines, [line.content for line in lines])
311                objects += list(sorted(lines + [obj], key=lambda o: (-o.bbox.y, o.bbox.x)))
312        return objects

122        """
123        Coalesce the characters in the area and predicate into lines.
124
125        1. Every character in the area is filtered by the `predicate`.
126        2. Character orientation is split into horizontal (left->right) and
127           vertical (bottom->top) character lines sorted by x or y position.
128           Lines containing only whitespace are discarded.
129        3. Overlapping character lines are merged into sub- and superscript
130           using `rtol * max(current_line.height, next_line.height)` as the
131           tolerance for checking if the lines overlap.
132        4. The characters in the merged lines are re-sorted by origin.
133
134        :param area: Area to search for characters.
135        :param predicate: Function to discard characters in the area or include all by default.
136        :param rtol: Relative tolerance to separate lines vertically or use `sc` spacing by default.
137        :return: A list of character lines sorted by x or y position.
138        """
139        if rtol is None: rtol = self._spacing["sc"]
140        # Split all chars into lines based on rounded origin
141        origin_lines_y = defaultdict(list)
142        origin_lines_x = defaultdict(list)
143        for char in self.chars_in_area(area):
144            # Ignore all characters we don't want
145            if predicate is not None and not predicate(char):
146                continue
147            cunicode = self._unicode_filter(char.unicode)
148            if cunicode is None: continue
149            char.unicode = cunicode
150            if char.unicode < 32 and char.unicode not in {0xa}:
151                continue
152            # Ignore characters without width that are not spaces
153            if not char.width and char.unicode not in {0xa, 0xd, 0x20}:
154                _LOGGER.error(f"Unknown char width for {char}: {char.bbox}")
155            # Split up the chars depending on the orientation
156            if 45 < char.rotation <= 135 or 225 < char.rotation <= 315:
157                origin_lines_x[round(char.origin.x, 1)].append(char)
158            elif char.rotation <= 45 or 135 < char.rotation <= 225 or 315 < char.rotation:

160            else:
