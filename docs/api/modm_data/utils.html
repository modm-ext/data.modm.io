261    def __init__(self, v0, v1, obj=None):
262        if v0 > v1:
263            v0, v1 = v1, v0

269    def overlaps(self, o0, o1, atol=0) -> bool:
270        if o0 > o1:
271            o0, o1 = o1, o0
272        # if reg top is lower then o0
273        if (self.v1 + atol) <= o0:
274            return False
275        # if reg bottom is higher than o1
276        if o1 <= (self.v0 - atol):
277            return False
278        return True
